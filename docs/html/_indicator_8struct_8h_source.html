38 struct IndicatorCalculateCache {

59  IndicatorCalculateCache(int _num_buffers = 0, int _buffers_size = 0) {
60  prev_calculated = 0;
61  num_buffers = _num_buffers;
62 
63  Resize(_buffers_size);
64  }
65 
69  void Resize(int _buffers_size) {
70  static int increase = 65536;
71  switch (num_buffers) {
72  case 5:
73  ArrayResize(buffer5, _buffers_size, (_buffers_size - _buffers_size % increase) + increase);
74  case 4:
75  ArrayResize(buffer4, _buffers_size, (_buffers_size - _buffers_size % increase) + increase);
76  case 3:
77  ArrayResize(buffer3, _buffers_size, (_buffers_size - _buffers_size % increase) + increase);
78  case 2:
79  ArrayResize(buffer2, _buffers_size, (_buffers_size - _buffers_size % increase) + increase);
80  case 1:
81  ArrayResize(buffer1, _buffers_size, (_buffers_size - _buffers_size % increase) + increase);
82  }
83  }
84 
88  double GetValue(int _buffer_index, int _shift) {
89  switch (_buffer_index) {
90  case 1:
91  return buffer1[ArraySize(buffer1) - 1 - _shift];
92  case 2:
93  return buffer2[ArraySize(buffer2) - 1 - _shift];
94  case 3:
95  return buffer3[ArraySize(buffer3) - 1 - _shift];
96  case 4:
97  return buffer4[ArraySize(buffer4) - 1 - _shift];
98  case 5:
99  return buffer5[ArraySize(buffer5) - 1 - _shift];
100  }
101  return DBL_MIN;
102  }
