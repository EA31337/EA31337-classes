54  bool GetPivots(ENUM_PP_TYPE _type, float &_pp, float &_r1, float &_r2, float &_r3, float &_r4, float &_s1, float &_s2,
55  float &_s3, float &_s4) {
56  float _range = GetRange();
57  switch (_type) {
58  case PP_CAMARILLA:
59  // A set of eight very probable levels which resemble support and resistance values for a current trend.
60  _pp = GetPivot();
61  _r1 = (float)(close + _range * 1.1 / 12);
62  _r2 = (float)(close + _range * 1.1 / 6);
63  _r3 = (float)(close + _range * 1.1 / 4);
64  _r4 = (float)(close + _range * 1.1 / 2);
65  _s1 = (float)(close - _range * 1.1 / 12);
66  _s2 = (float)(close - _range * 1.1 / 6);
67  _s3 = (float)(close - _range * 1.1 / 4);
68  _s4 = (float)(close - _range * 1.1 / 2);
69  break;
70  case PP_CLASSIC:
71  _pp = GetPivot();
72  _r1 = (2 * _pp) - low; // R1 = (H - L) * 1.1 / 12 + C (1.0833)
73  _r2 = _pp + _range; // R2 = (H - L) * 1.1 / 6 + C (1.1666)
74  _r3 = _pp + _range * 2; // R3 = (H - L) * 1.1 / 4 + C (1.25)
75  _r4 = _pp + _range * 3; // R4 = (H - L) * 1.1 / 2 + C (1.5)
76  _s1 = (2 * _pp) - high; // S1 = C - (H - L) * 1.1 / 12 (1.0833)
77  _s2 = _pp - _range; // S2 = C - (H - L) * 1.1 / 6 (1.1666)
78  _s3 = _pp - _range * 2; // S3 = C - (H - L) * 1.1 / 4 (1.25)
79  _s4 = _pp - _range * 3; // S4 = C - (H - L) * 1.1 / 2 (1.5)
80  break;
81  case PP_FIBONACCI:
82  _pp = GetPivot();
83  _r1 = (float)(_pp + 0.382 * _range);
84  _r2 = (float)(_pp + 0.618 * _range);
85  _r3 = _pp + _range;
86  _r4 = _r1 + _range; // ?
87  _s1 = (float)(_pp - 0.382 * _range);
88  _s2 = (float)(_pp - 0.618 * _range);
89  _s3 = _pp - _range;
90  _s4 = _s1 - _range; // ?
91  break;
92  case PP_FLOOR:
93  // Most basic and popular type of pivots used in Forex trading technical analysis.
94  _pp = GetPivot(); // Pivot (P) = (H + L + C) / 3
95  _r1 = (2 * _pp) - low; // Resistance (R1) = (2 * P) - L
96  _r2 = _pp + _range; // R2 = P + H - L
97  _r3 = high + 2 * (_pp - low); // R3 = H + 2 * (P - L)
98  _r4 = _r3;
99  _s1 = (2 * _pp) - high; // Support (S1) = (2 * P) - H
100  _s2 = _pp - _range; // S2 = P - H + L
101  _s3 = low - 2 * (high - _pp); // S3 = L - 2 * (H - P)
102  _s4 = _s3; // ?
103  break;
104  case PP_TOM_DEMARK:
105  // Tom DeMark's pivot point (predicted lows and highs of the period).
106  _pp = GetPivotDeMark();
107  _r1 = (2 * _pp) - low; // New High = X / 2 - L.
108  _r2 = _pp + _range;
109  _r3 = _r1 + _range;
110  _r4 = _r2 + _range; // ?
111  _s1 = (2 * _pp) - high; // New Low = X / 2 - H.
112  _s2 = _pp - _range;
113  _s3 = _s1 - _range;
114  _s4 = _s2 - _range; // ?
115  break;
116  case PP_WOODIE:
117  // Woodie's pivot point are giving more weight to the Close price of the previous period.
118  // They are similar to floor pivot points, but are calculated in a somewhat different way.
119  _pp = GetWeighted(); // Pivot (P) = (H + L + 2 * C) / 4
120  _r1 = (2 * _pp) - low; // Resistance (R1) = (2 * P) - L
121  _r2 = _pp + _range; // R2 = P + H - L
122  _r3 = _r1 + _range;
123  _r4 = _r2 + _range; // ?
124  _s1 = (2 * _pp) - high; // Support (S1) = (2 * P) - H
125  _s2 = _pp - _range; // S2 = P - H + L
126  _s3 = _s1 - _range;
127  _s4 = _s2 - _range; // ?
128  break;
129  }
130  return _r4 > _r3 && _r3 > _r2 && _r2 > _r1 && _r1 > _pp && _pp > _s1 && _s1 > _s2 && _s2 > _s3 && _s3 > _s4;
131  }
